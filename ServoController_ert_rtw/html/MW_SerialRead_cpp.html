78extern "C" void fillDataBuffer(const uint8_T port,const uint16_t DataSizeInBytes,char *data,uint8_T *status)



79{



80    char databuffer[2*DataSizeInBytes];    



81    (*read_func_ptr[port])(databuffer,DataSizeInBytes);



82    for(int ii=0;ii< DataSizeInBytes;ii++)



83    {



84        *data++ = databuffer[ii];



85    }



86    *status = 1;



87}

91extern "C" void MW_Serial_read(const uint8_T port,const uint16_t DataSizeInBytes, char *data,uint8_T *status)



92{



93    



94    int dataSizeinBuffer;



95    char tempdatabuffer[64];    



96    



97    



98    /* Read the size of data available in the Arduino Serial buffer */



99    dataSizeinBuffer = (*available_func_ptr[port])();



100    if (dataSizeinBuffer == DataSizeInBytes)



101    {



102        /* If the size of data in buffer == required data size, data=received data and status =1 */



103        fillDataBuffer(port,DataSizeInBytes,data,status);



104    }



105    else if ( dataSizeinBuffer >= DataSizeInBytes)



106    {



107        if (dataSizeinBuffer <= 2*DataSizeInBytes)



108        {



109            /* If the size of data in buffer > required data size and < 2*required data size, data=received data(of the rqquired size) and status =1 */



110            fillDataBuffer(port,DataSizeInBytes,data,status);



111        }



112        else



113        {



114            /* If the size of data in buffer >  2*required data size, trash data till size of data in buffer < 2*required data size.data=received data and status =1 */



115            (*read_func_ptr[port])(tempdatabuffer,(((dataSizeinBuffer/DataSizeInBytes)-1)*DataSizeInBytes));



116            fillDataBuffer(port,DataSizeInBytes,data,status);



117        }



118    }



119    else



120    {



121        /* If the size of data in buffer < required data size, data=0's  and status =0 */



122        for(int ii=0;ii< DataSizeInBytes;ii++)



123        {



124            *data++ = 0;



125        }



126        *status = 0;



127    }



128}
